 * @param {number} x viewport x

export function bubbleAt(x, y) {
  const bubbles = getBubbles();
  for (const b of bubbles) {
    if ((x - b.cx) ** 2 + (y - b.cy) ** 2 <= b.r * b.r) return b;

  return null;
}
